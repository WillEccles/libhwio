120 static int set_interface_attribs_rs485(int fd, Serial_Params* params) {
121  struct serial_rs485 rs485conf = {0};
122 
123  // enable RS485 mode
124  rs485conf.flags |= SER_RS485_ENABLED;
125 
126  // default to RTS inactive when sending
127  rs485conf.flags &= ~(SER_RS485_RTS_ON_SEND);
128  // change if user specified
129  if (params->rs485_RTSHighOnSend) {
130  rs485conf.flags |= SER_RS485_RTS_ON_SEND;
131  }
132 
133  // default to RTS inactive after sending
134  rs485conf.flags &= ~(SER_RS485_RTS_AFTER_SEND);
135  // change if user specified
136  if (params->rs485_RTSHighAfterSend) {
137  rs485conf.flags |= SER_RS485_RTS_AFTER_SEND;
138  }
139 
140  // don't receive while sending
141  rs485conf.flags &= ~(SER_RS485_RX_DURING_TX);
142 
143  if (0 != ioctl(fd, TIOCSRS485, &rs485conf)) {
144  return -1;
145  }
146 
147  return set_interface_attribs_raw(fd, get_baud_value(params));
148 }
