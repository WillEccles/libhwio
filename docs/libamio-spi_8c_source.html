20SPI_Handle SPI_open(int bus, int cs, SPI_Params* params) {
21 if (bus < 0 || cs < 0) return NULL;

35SPI_Handle SPI_open_path(const char* path, SPI_Params* params) {
36 spidevice* dev = (spidevice*)malloc(sizeof(spidevice));
37 if (dev == NULL) return NULL;
38
39 dev->bus = -1;
40 dev->cs = -1;
41
42 if (EXIT_SUCCESS != access(path, F_OK)) {
43 free(dev);
44 return NULL;
45 }
46
47 dev->fd = open(path, O_SYNC | O_RDWR | O_CLOEXEC);
48
49 if (dev->fd < 0) {
50 free(dev);
51 return NULL;
52 }
53
54 // set the bits per word
55 if (params->wordsize == 0) {
56 goto errorafterinit;
57 }
58
59 if (-1 == ioctl(dev->fd, SPI_IOC_WR_BITS_PER_WORD, &(params->wordsize))) {
60 goto errorafterinit;
61 }
62
63 // set the bitrate
64 if (params->bitrate == 0) {
65 goto errorafterinit;
66 }
67
68 // set the SPI speed
69 if (-1 == ioctl(dev->fd, SPI_IOC_WR_MAX_SPEED_HZ, &(params->bitrate))) {
70 goto errorafterinit;
71 }
72
73 // set the SPI mode
74 if (-1 == ioctl(dev->fd, SPI_IOC_WR_MODE32, &(params->mode))) {
75 goto errorafterinit;
76 }
77
78 // if we have been successful the whole time,
79 // we can return a handle to the SPI device
80 return (SPI_Handle)dev;
81
82errorafterinit:
83 close(dev->fd);
84 free(dev);
85 return NULL;
86}

88int SPI_transfer(SPI_Handle handle, SPI_Transaction* transaction) {
89 if (handle == NULL || transaction == NULL || transaction->count < 0) {
90 return EXIT_FAILURE;
91 }
92
93 if (transaction->count == 0) {
94 // just call this successful, since technically we didn't fail at anything
95 return EXIT_SUCCESS;
96 }
97
98 spidevice* dev = (spidevice*)handle;
99
100 // we don't have to do anything special to accomodate for rxBuf being NULL,
101 // since making .rx_buf == NULL already does this for us
102 struct spi_ioc_transfer msg = (struct spi_ioc_transfer){
103 .rx_buf = (unsigned long)transaction->rxBuf,
104 .tx_buf = (unsigned long)transaction->txBuf,
105 .len = transaction->count,
106 .speed_hz = transaction->speed_hz,
107 };
108
109 int r = ioctl(dev->fd, SPI_IOC_MESSAGE(1), &msg);
110 if (r < 0) {
111 return EXIT_FAILURE;
112 }
113
114 return EXIT_SUCCESS;
115}
