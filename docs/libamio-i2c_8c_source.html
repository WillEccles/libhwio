62 
63  if (dev->fd < 0) {
64  return EXIT_FAILURE;
65  }
66 
67  if ((transaction->readCount && transaction->readBuf == NULL)
68  || (transaction->writeCount && transaction->writeBuf == NULL)) {
69  return EXIT_FAILURE;
70  }
71 
72  // set timeout
73  if (EXIT_SUCCESS != set_timeout(dev, transaction->timeout / 10)) {
74  return EXIT_FAILURE;
75  }

83  .flags = 0,
84  .len = transaction->writeCount,
85  .buf = transaction->writeBuf,
86  };
87 
88  if (EXIT_FAILURE == i2c_rdwr(dev, &msg)) {
89  return EXIT_FAILURE;
90  }
91  }
92

95  .addr = transaction->slaveAddress,

97  .len = transaction->readCount,
98  .buf = transaction->readBuf,
99  };
